PREFIX : <http://dublincore.org/profile>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX rdfs: <http://www.w3.org/1999/02/22-rdf-syntax-ns>

start = @<Profile>

<Profile> {               # A Profile has:
  :EntityShape @<EntityShape>* ;
  :Property @<PropertyShape>+         # one or more PropertyShape(s) 
}

<EntityShape> {
    :EntityShapeID IRI;    #
    :Property @<PropertyShape>+ 
    }

<PropertyShape> {                   # A Property has:
 :PropertyID IRI ;  #   an identifier, IRI or string
 :PropertyLabel [xsd:string rdfs:langString]* ; # an optional label 
 :Mandatory  ["y" "n"] ; 
 :Repeatable ["y" "n"] ;
 :ValueType ["IRI" "LITERAL" "BNODE"] ; # The general value type
 :ValueDataType IRI ; # any identified value type; preference to xsd types
 :ValueConstraint xsd:string ; # additional value constraints
 :EntityShapeReference Nonliteral ; # A reference to an entity in the profile
 :Annotation xsd:string ; # any information needed about property + value 
}
